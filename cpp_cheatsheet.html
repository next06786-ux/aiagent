1. 位运算基础

a & b    // 按位与：两位都为1才为1；常用：判断奇偶、取特定位
a | b    // 按位或：有1则为1；常用：置位

三、数学算法

1. 最大公约数 gcd

int gcd(int a, int b) {
    return b == 0 ? a : gcd(b, a % b);
}
// 思路：辗转相除法。gcd(a,b)=gcd(b,a%b)，b为0时a就是答案。
// 作用：求a和b的最大公约数

2. 最小公倍数 lcm

LL lcm(LL a, LL b) {
    return a / gcd(a, b) * b;
}
// 思路：lcm = a*b/gcd(a,b)，先除再乘防止溢出。
// 作用：求a和b最小公倍数

3. 快速幂取模

LL qpow(LL a, LL b, LL mod) {
    LL res = 1;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}
// 思路：将指数b写成二进制，逐位处理。b的第k位为1则乘上a^(2^k)。
// 每轮a平方，b右移一位，O(log b)次乘法。
// 作用：快速求 a^b % mod；a：底数；b：指数；mod：模数

4. 素数判定

bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i++)
        if (n % i == 0) return false;
    return true;
}
// 思路：若n有因子，必有一个≤√n。只需枚举到√n即可，O(√n)。
// 作用：判断n是否为素数

5. 埃氏筛

const int MAX = 1e5 + 10;
bool isPrime[MAX];
void sieve(int n) {
    memset(isPrime, 1, sizeof(isPrime));
    isPrime[0] = isPrime[1] = 0;
    for (int i = 2; i <= n; i++)
        if (isPrime[i])
            for (int j = 2*i; j <= n; j += i)
                isPrime[j] = 0;
}
// 思路：每找到一个素数i，把它所有倍数标记为合数。
// 同一个合数会被多个素因子重复标记，O(n log log n)。
// 作用：筛出1~n所有素数；n：上限

6. 欧拉筛（线性筛）

const int MAXN = 1e6 + 10;
int primes[MAXN], cnt;
bool vis[MAXN];
void euler(int n) {
    for (int i = 2; i <= n; i++) {
        if (!vis[i]) primes[cnt++] = i;
        for (int j = 0; primes[j] * i <= n; j++) {
            vis[primes[j] * i] = true;
            if (i % primes[j] == 0) break;  // 保证每个合数只被最小质因子标记一次
        }
    }
}
// 思路：每个合数只被其最小质因子筛掉，保证每个数只被标记一次，O(n)。
// 关键：i%primes[j]==0时break，此后primes[j]不再是primes[j]*i的最小质因子。
// 作用：O(n)筛素数，速度最快；n：上限

7. 十进制转D进制

void toBase(int n, int D) {
    if (n == 0) { printf("0"); return; }
    int stk[105], top = 0;
    while (n) stk[++top] = n % D, n /= D;
    while (top) printf("%d", stk[top--]);
}
// 思路：反复对D取余得到各位（低位先出），用栈逆序输出得到高位在前的结果。
// 作用：十进制n转D进制输出；n：十进制数；D：目标进制

8. 组合数 C(n,k)

LL C(int n, int k) {
    if (k < 0 || k > n) return 0;
    LL res = 1;
    for (int i = 1; i <= k; i++)
        res = res * (n - i + 1) / i;
    return res;
}
// 思路：C(n,k) = n*(n-1)*...*(n-k+1) / k!，逐步乘除保持整除性。
// 注意：每步先乘再除，利用连续整数必有一个被i整除的性质保证整除。
// 作用：计算组合数C(n,k)；n较大时需改用预处理+逆元版本

四、排序与二分

1. 快速排序

int Partition(int a[], int l, int r) {
    int pivot = a[l];
    while (l < r) {
        while (l < r && a[r] >= pivot) r--;
        a[l] = a[r];
        while (l < r && a[l] <= pivot) l++;
        a[r] = a[l];
    }
    a[l] = pivot;
    return l;
}
void qsort(int a[], int l, int r) {
    if (l < r) {
        int pos = Partition(a, l, r);
        qsort(a, l, pos - 1);
        qsort(a, pos + 1, r);
    }
}
// 思路：选基准pivot，双指针从两端向中间扫，把小的放左边大的放右边。
// 分区后pivot归位，递归处理左右两段。平均O(n log n)，最坏O(n²)。
// 作用：快排；a[]：数组；l：左边界；r：右边界

2. 二分查找（第一个 >= x）

int lower(int a[], int n, int x) {
    int l = 1, r = n;
    while (l < r) {
        int mid = (l + r) / 2;
        if (a[mid] >= x) r = mid;
        else l = mid + 1;
    }
    return l;
}
// 思路：维护答案在[l,r]内。mid满足条件则答案在左半（r=mid），否则在右半（l=mid+1）。
// 循环结束时l==r即为答案。关键：r=mid而非r=mid-1，保证不漏掉答案。
// 作用：升序数组中找第一个>=x的位置

3. 二分查找（第一个 > x）

int upper(int a[], int n, int x) {
    int l = 1, r = n;
    while (l < r) {
        int mid = (l + r) / 2;
        if (a[mid] > x) r = mid;
        else l = mid + 1;
    }
    return l;
}
// 思路：同上，仅将条件改为严格大于x。两种二分模板的区别只在判断条件。

五、搜索（DFS / BFS / 全排列）

1. DFS 连通块

vector<int> G[1005];
bool vis[1005];
void dfs(int u) {
    vis[u] = true;
    for (int v : G[u])
        if (!vis[v]) dfs(v);
}
// 思路：从u出发，标记u已访问，递归访问所有未访问的邻居。
// 一次dfs(u)访问完u所在的整个连通块，统计调用次数即连通块数。
// 作用：图遍历、连通块统计；u：当前节点；G：邻接表；vis：访问标记

2. BFS 无权最短路

void bfs(int start) {
    queue<int> q;
    q.push(start); vis[start] = 1;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : G[u])
            if (!vis[v]) vis[v] = 1, q.push(v);
    }
}
// 思路：按层扩展，先入队的先处理，保证第一次到达某节点时路径最短。
// 用dist[]记录距离：dist[v]=dist[u]+1，初始dist[start]=0。
// 作用：求最短路、连通块；start：起点

3. 全排列

int a[10], vis[10], n;
void dfs(int k) {
    if (k == n) { /* 输出排列 */ return; }
    for (int i = 1; i <= n; i++)
        if (!vis[i]) { vis[i]=1; a[k]=i; dfs(k+1); vis[i]=0; }
}
// 思路：逐位填数，第k位枚举所有未用过的数字，填入后递归填下一位。
// 回溯时撤销选择（vis[i]=0），保证每条路径独立。共n!种排列。
// 作用：枚举1~n全排列；k：当前填第k位

4. 子集枚举（回溯）

int a[20], n;
void dfs(int k) {
    if (k == n) { /* 处理当前子集 */ return; }
    a[k] = 0; dfs(k+1);  // 不选第k个元素
    a[k] = 1; dfs(k+1);  // 选第k个元素
}
// 思路：对每个元素做二选一（选/不选），递归到底时a[]记录了一个子集。
// 共2^n个叶子节点，对应2^n个子集。也可用位运算枚举（更快）。
// 作用：枚举所有2^n个子集；k：当前考虑第k个元素

5. 二维网格 BFS（最短路）

int dx[] = {0,0,1,-1};  // 四方向偏移量
int dy[] = {1,-1,0,0};
int dist[105][105];
void bfs(int sx, int sy, int n, int m) {
    memset(dist, -1, sizeof(dist));
    queue<pair<int,int>> q;
    dist[sx][sy] = 0; q.push({sx, sy});
    while (!q.empty()) {
        auto [x, y] = q.front(); q.pop();
        for (int d = 0; d < 4; d++) {
            int nx = x+dx[d], ny = y+dy[d];
            if (nx>=0 && nx<n && ny>=0 && ny<m && dist[nx][ny]==-1)
                dist[nx][ny] = dist[x][y]+1, q.push({nx,ny});
        }
    }
}
// 思路：将二维坐标(x,y)作为BFS节点，用dx/dy数组枚举四个方向。
// dist==-1表示未访问，BFS保证第一次到达时距离最短。
// 多源BFS：把所有起点同时入队，dist初始化为0即可。
// sx/sy：起点坐标；n/m：网格行列数；nx/ny：下一格坐标

6. 记忆化搜索（DFS+memo）

int memo[1005][1005];
int dfs(int i, int j) {
    if (memo[i][j] != -1) return memo[i][j];
    // ... 递归计算 ...
    return memo[i][j] = res;
}
// 思路：DFS天然表达递推关系，memo数组缓存已算过的状态避免重复计算。
// 本质是自顶向下的DP，与自底向上的迭代DP等价，写法更直观。
// 初始化：memset(memo, -1, sizeof(memo));
// i/j：状态参数；res：当前状态的答案

六、图论

1. 并查集

int fa[1005];
void init(int n) { for (int i=1;i<=n;i++) fa[i]=i; }
int find(int x) { return fa[x]==x ? x : fa[x]=find(fa[x]); }
void unite(int x, int y) { x=find(x); y=find(y); if(x!=y) fa[x]=y; }
// 思路：每个集合用一棵树表示，fa[x]是x的父节点，根节点是集合代表。
// find()路径压缩：查找时把沿途所有节点直接挂到根，后续查找O(1)。
// unite()：找到两个根，把一个根挂到另一个根下完成合并。
// 作用：集合合并、判断连通；n：元素总数；x/y：节点编号

2. Dijkstra 单源最短路

typedef pair<int,int> pii;
vector<pii> G[1005];
int d[1005];
void dijkstra(int s, int n) {
    memset(d, 0x3f, sizeof(d));
    priority_queue<pii,vector<pii>,greater<pii>> q;
    d[s] = 0; q.push({0, s});
    while (!q.empty()) {
        int u = q.top().second; q.pop();
        for (auto &e : G[u]) {
            int v = e.first, w = e.second;
            if (d[v] > d[u] + w)
                d[v] = d[u] + w, q.push({d[v], v});
        }
    }
}
// 思路：贪心。每次从优先队列取出当前距离最小的节点u（已确定最短路），
// 用u松弛其所有邻居v：若d[u]+w < d[v]则更新并入队。
// 负权边会破坏贪心性质，不能用Dijkstra。
// 作用：带权图单源最短路（边权非负）；s：起点；G[u]：{v, w}

3. Floyd 全源最短路

int d[105][105];
void floyd(int n) {
    for (int k=1;k<=n;k++)
    for (int i=1;i<=n;i++)
    for (int j=1;j<=n;j++)
        d[i][j] = min(d[i][j], d[i][k]+d[k][j]);
}
// 思路：DP。d[i][j]表示只经过前k个节点作中转时i到j的最短路。
// 枚举中转点k，更新所有点对(i,j)：经过k是否更短。O(n³)。
// 初始化：d[i][i]=0，有边(u,v,w)则d[u][v]=w，其余INF。
// 作用：任意两点最短路；n：点数；d[i][j]：距离

4. 拓扑排序

vector<int> G[1005];
int in[1005];
bool topo(int n) {
    queue<int> q; int cnt=0;
    for (int i=1;i<=n;i++) if(in[i]==0) q.push(i);
    while (!q.empty()) {
        int u=q.front(); q.pop(); cnt++;
        for (int v : G[u]) if(--in[v]==0) q.push(v);
    }
    return cnt == n;
}
// 思路：入度为0的节点没有前驱，可以最先输出。输出后删去其出边，
// 新产生的入度为0的节点继续入队。若最终输出了n个节点则无环（DAG）。
// 作用：判断DAG、拓扑序；n：点数；in：入度

5. Bellman-Ford（含负权边）

struct Edge { int u,v,w; } edges[10005];
int d[1005];
bool bellman(int s, int n, int m) {
    memset(d, 0x3f, sizeof(d)); d[s]=0;
    for (int i=0;i<n-1;i++)           // 松弛n-1轮
        for (int j=0;j<m;j++) {
            int u=edges[j].u, v=edges[j].v, w=edges[j].w;
            if (d[u]+w < d[v]) d[v]=d[u]+w;
        }
    for (int j=0;j<m;j++)             // 检测负环
        if (d[edges[j].u]+edges[j].w < d[edges[j].v]) return false;
    return true;
}
// 思路：最短路最多经过n-1条边，所以松弛n-1轮后必收敛。
// 每轮对所有边尝试松弛。第n轮若还能松弛，说明存在负环。O(nm)。
// 作用：含负权边最短路；返回false表示有负环；s：起点；n：点数；m：边数

6. Kruskal 最小生成树

struct Edge { int u,v,w; } edges[10005];
int fa[1005];
int find(int x) { return fa[x]==x?x:fa[x]=find(fa[x]); }
int kruskal(int n, int m) {
    sort(edges, edges+m, [](Edge a,Edge b){ return a.w<b.w; });
    for (int i=1;i<=n;i++) fa[i]=i;
    int res=0, cnt=0;
    for (int i=0;i<m;i++) {
        int u=find(edges[i].u), v=find(edges[i].v);
        if (u!=v) { fa[u]=v; res+=edges[i].w; cnt++; }
    }
    return cnt==n-1 ? res : -1;
}
// 思路：贪心。按边权从小到大排序，依次考虑每条边：
// 若两端点不在同一连通分量（用并查集判断），则加入生成树，否则跳过（会成环）。
// 选够n-1条边后结束。O(m log m)。
// 作用：求最小生成树总权值；res：总权值；cnt：已选边数；-1表示不连通

7. 树的直径（两次BFS/DFS）

int farthest, maxDist;
void bfs_tree(int s, int n) {
    vector<int> dist(n+1, -1);
    queue<int> q; dist[s]=0; q.push(s);
    while (!q.empty()) {
        int u=q.front(); q.pop();
        for (int v : G[u]) if(dist[v]==-1)
            dist[v]=dist[u]+1, q.push(v);
    }
    farthest=max_element(dist.begin(),dist.end())-dist.begin();
    maxDist=*max_element(dist.begin(),dist.end());
}
// 思路：结论——从任意点出发BFS/DFS到最远点u，再从u出发到最远点v，u-v路径即为直径。
// 用法：bfs_tree(1,n) 得到端点u=farthest，再 bfs_tree(u,n) 得到直径长度maxDist。
// 作用：求树的直径（最长路径）；s：起点；n：点数

七、动态规划 DP

1. 01 背包

int dp[1005];
void zeroOne(int V, int v[], int w[], int n) {
    memset(dp, 0, sizeof(dp));
    for (int i=0;i<n;i++)
        for (int j=V;j>=v[i];j--)
            dp[j] = max(dp[j], dp[j-v[i]]+w[i]);
}
// 思路：dp[j]表示容量恰好/最多为j时的最大价值。
// 内层倒序枚举：保证每件物品只被选一次（若正序则同一件物品可被多次选）。
// 转移：选第i件（dp[j-v[i]]+w[i]）vs 不选（dp[j]），取最大。
// V：背包容量；v[]体积；w[]价值；n物品数

2. 完全背包

void complete(int V, int v[], int w[], int n) {
    memset(dp, 0, sizeof(dp));
    for (int i=0;i<n;i++)
        for (int j=v[i];j<=V;j++)
            dp[j] = max(dp[j], dp[j-v[i]]+w[i]);
}
// 思路：与01背包唯一区别是内层正序枚举。
// 正序时dp[j-v[i]]已经是本轮更新过的值，相当于允许再次选同一件物品。
// 作用：物品可多选

3. LIS 最长不下降子序列

int LIS(int a[], int n) {
    int dp[1005], res=0;
    for (int i=0;i<n;i++) {
        dp[i]=1;
        for (int j=0;j<i;j++)
            if (a[j]<=a[i]) dp[i]=max(dp[i],dp[j]+1);
        res=max(res,dp[i]);
    }
    return res;
}
// 思路：dp[i]表示以a[i]结尾的最长不下降子序列长度。
// 枚举i之前所有j，若a[j]<=a[i]则a[i]可接在a[j]后面，dp[i]=max(dp[i],dp[j]+1)。
// O(n²)，大数据量用下面的O(n log n)版本。

4. LCS 最长公共子序列

int LCS(char s1[], char s2[], int n, int m) {
    int dp[1005][1005];
    for (int i=1;i<=n;i++)
    for (int j=1;j<=m;j++)
        if (s1[i]==s2[j]) dp[i][j]=dp[i-1][j-1]+1;
        else dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
    return dp[n][m];
}
// 思路：dp[i][j]表示s1前i个字符与s2前j个字符的LCS长度。
// 若末尾字符相同，LCS长度=dp[i-1][j-1]+1；否则从去掉一个末尾字符的两种情况取最大。

5. LIS O(n log n) 二分优化

int LIS_fast(int a[], int n) {
    vector<int> tails;
    for (int i=0;i<n;i++) {
        auto it = lower_bound(tails.begin(), tails.end(), a[i]);
        if (it == tails.end()) tails.push_back(a[i]);
        else *it = a[i];
    }
    return tails.size();
}
// 思路：tails[k]维护长度为k+1的LIS末尾元素的最小值（贪心：末尾越小越好接后续元素）。
// 对每个a[i]，二分找tails中第一个>=a[i]的位置并替换（严格递增用lower_bound）。
// 找不到则追加，tails长度即为LIS长度。注意tails不是实际的LIS序列。
// 改lower_bound为upper_bound可求不下降子序列

6. 区间 DP

int dp[505][505];
void intervalDP(int n) {
    for (int len=2;len<=n;len++)
    for (int i=1;i+len-1<=n;i++) {
        int j=i+len-1;
        dp[i][j] = INF;

7. 树形 DP（以树的最大独立集为例）

int dp[1005][2];
void dfs(int u, int fa) {

    for (int v : G[u]) {
        if (v==fa) continue;
        dfs(v, u);
        dp[u][0] += max(dp[v][0], dp[v][1]);

8. 数字 DP 模板

int digit[20], dp[20][2];
int dfs(int pos, bool tight, bool started) {
    if (pos < 0) return 1;
    if (dp[pos][tight] != -1) return dp[pos][tight];
    int limit = tight ? digit[pos] : 9;
    int res = 0;
    for (int d=0; d<=limit; d++)
        res += dfs(pos-1, tight && (d==limit), started || d);
    return dp[pos][tight] = res;
}
// 思路：逐位填数字，tight标记当前位是否受上界约束（若前面每位都取最大值则tight=true）。
// tight=true时当前位最多填digit[pos]，否则可填0~9。
// 用记忆化避免重复计算相同(pos,tight)状态。
// 统计[1,n]中满足条件的数：先把n各位存入digit[]，再调用dfs(最高位,true,false)。
// pos：当前位；tight：是否受上界限制；started：是否已开始填数（处理前导零）

9. 背包变种：多重背包

void multiPack(int V, int v[], int w[], int cnt[], int n) {
    vector<pair<int,int>> items;
    for (int i=0;i<n;i++) {
        int c=cnt[i], k=1;
        while (c>0) {
            int t=min(k,c);
            items.push_back({t*v[i], t*w[i]});
            c-=t; k*=2;
        }
    }
    memset(dp,0,sizeof(dp));
    for (auto &[vi,wi] : items)
        for (int j=V;j>=vi;j--)
            dp[j]=max(dp[j], dp[j-vi]+wi);
}
// 思路：二进制拆分。把cnt[i]件物品拆成1,2,4,...,余数 几组，
// 这些组合可以凑出0~cnt[i]的任意数量，转化为01背包处理。
// 比朴素O(V*Σcnt)快，达到O(V*Σlog(cnt))。
// V：容量；v[]：体积；w[]：价值；cnt[]：每种数量；n：种数

2. KMP 字符串匹配

int nxt[1005];
void getNext(char p[]) {
    int n=strlen(p),i=0,j=-1; nxt[0]=-1;
    while (i<n) {
        if (j==-1 || p[i]==p[j]) nxt[++i]=++j;
        else j = nxt[j];

int kmp(char s[], char p[]) {

    while (i<n) {
        if (j==-1 || s[i]==p[j]) i++,j++;
        else j = nxt[j];
        if (j==m) cnt++,j=nxt[j];
    }
    return cnt;
}
// 思路：nxt[i]表示p[0..i-1]的最长相等前后缀长度（失配时j跳到nxt[j]而非0）。

3. 字符串哈希

const LL BASE = 131, MOD1 = 1e9+7, MOD2 = 1e9+9;
LL h[1005], pw[1005];
void buildHash(char s[], int n) {
    pw[0]=1;
    for (int i=1;i<=n;i++) {
        h[i] = (h[i-1]*BASE + s[i]) % MOD1;
        pw[i] = pw[i-1]*BASE % MOD1;
    }
}
LL getHash(int l, int r) {
    return (h[r] - h[l-1]*pw[r-l+1] % MOD1 + MOD1*2) % MOD1;
}
// 思路：把字符串看成BASE进制数，h[i]是前i个字符的哈希值（类似前缀和）。
// 子串[l,r]的哈希 = h[r] - h[l-1]*BASE^(r-l+1)，O(1)取子串哈希。
// 两个子串哈希相等则大概率内容相同（双模哈希可降低冲突概率）。
// s：字符串（1-indexed）；n：长度；l/r：区间端点

4. 最长回文子串（Manacher）

int manacher(char s[], int n) {
    char t[2005]; int m=0;
    t[m++]='#';
    for (int i=0;i<n;i++) t[m++]=s[i], t[m++]='#';
    int p[2005]={}, c=0, r=0, ans=0;
    for (int i=0;i<m;i++) {
        p[i] = r>i ? min(p[2*c-i], r-i) : 1;
        while (i-p[i]>=0 && i+p[i]<m && t[i-p[i]]==t[i+p[i]]) p[i]++;
        if (i+p[i]>r) r=i+p[i], c=i;
        ans=max(ans, p[i]-1);
    }
    return ans;
}
// 思路：插入#统一奇偶回文。利用已知最右回文[c-r,c+r]，

    sort(segs.begin(), segs.end());

    int l=segs[0].first, r=segs[0].second;
    for (auto &seg : segs) {
        if (seg.first <= r) r=max(r, seg.second);

int huffman(vector<int> a) {
    priority_queue<int,vector<int>,greater<int>> pq(a.begin(),a.end());
    int cost=0;
    while (pq.size()>1) {
        int x=pq.top(); pq.pop();
        int y=pq.top(); pq.pop();
        cost+=x+y; pq.push(x+y);
    }
    return cost;
}

5. 线段树（区间修改+区间查询）

struct Node { int sum, lazy; } tree[4005];
void build(int node, int l, int r) {
    if (l==r) { tree[node].sum=a[l]; return; }
    int mid=(l+r)/2;
    build(2*node,l,mid); build(2*node+1,mid+1,r);
    tree[node].sum=tree[2*node].sum+tree[2*node+1].sum;
}
void pushdown(int node, int l, int r) {
    if (!tree[node].lazy) return;
    int mid=(l+r)/2, lz=tree[node].lazy;
    tree[2*node].sum+=lz*(mid-l+1);   tree[2*node].lazy+=lz;
    tree[2*node+1].sum+=lz*(r-mid);   tree[2*node+1].lazy+=lz;
    tree[node].lazy=0;
}
void update(int node, int l, int r, int ql, int qr, int val) {
    if (ql<=l && r<=qr) { tree[node].sum+=val*(r-l+1); tree[node].lazy+=val; return; }
    pushdown(node,l,r);
    int mid=(l+r)/2;
    if (ql<=mid) update(2*node,l,mid,ql,qr,val);
    if (qr>mid)  update(2*node+1,mid+1,r,ql,qr,val);
    tree[node].sum=tree[2*node].sum+tree[2*node+1].sum;
}
int queryTree(int node, int l, int r, int ql, int qr) {
    if (ql<=l && r<=qr) return tree[node].sum;
    pushdown(node,l,r);
    int mid=(l+r)/2, res=0;
    if (ql<=mid) res+=queryTree(2*node,l,mid,ql,qr);
    if (qr>mid)  res+=queryTree(2*node+1,mid+1,r,ql,qr);
    return res;
}
// 思路：二叉树每个节点维护一段区间的信息（sum）和懒标记（lazy）。
// 区间修改时若当前节点区间被完全覆盖，打lazy标记不下传（延迟更新）。
// 访问子节点前先pushdown下传lazy，保证子节点信息正确。O(log n)修改和查询。
// node：节点编号（根为1，左子2*node，右子2*node+1）；l/r：节点管辖区间；ql/qr：操作区间

十二、数学补充

1. 扩展欧几里得

int exgcd(int a, int b, int &x, int &y) {
    if (!b) { x=1; y=0; return a; }
    int d=exgcd(b, a%b, y, x);
    y -= a/b*x;
    return d;
}
// 思路：在gcd递归过程中反向推导x,y。
// 递归到b=0时x=1,y=0（a*1+0*0=a）。回溯时利用 a*x+b*y=b*y+(a%b)*x 反推上层解。
// 作用：求ax+by=gcd(a,b)的一组整数解；x/y：输出参数；返回值：gcd(a,b)

2. 逆元（费马小定理，mod为质数）

LL inv(LL a, LL mod) { return qpow(a, mod-2, mod); }
// 思路：费马小定理：a^(p-1)≡1(mod p)，故a的逆元=a^(p-2) mod p。
// 仅当mod为质数且gcd(a,mod)=1时成立。用快速幂计算，O(log p)。
// 线性预处理逆元（1~n，O(n)）：
inv[1]=1;
for (int i=2;i<=n;i++) inv[i]=(mod-mod/i)*inv[mod%i]%mod;
// 推导：mod = (mod/i)*i + mod%i，两边取逆元化简得递推式。

3. 中国剩余定理（CRT）

LL crt(LL r[], LL m[], int k) {
    LL M=1, ans=0;
    for (int i=0;i<k;i++) M*=m[i];
    for (int i=0;i<k;i++) {
        LL Mi=M/m[i], ti=inv(Mi,m[i]);
        ans=(ans+r[i]*Mi%M*ti)%M;
    }
    return ans;
}
// 思路：M=所有模数之积，Mi=M/m[i]，ti=Mi在m[i]意义下的逆元。
// 每项r[i]*Mi*ti满足：对m[i]取模=r[i]，对其他m[j]取模=0。叠加即得解。
// 要求m[i]两两互质；r[]：余数；m[]：模数；k：方程数

typedef vector<vector<LL>> Mat;
Mat mul(Mat &A, Mat &B, int n) {

    for (int k=0;k<n;k++) if(A[i][k])

Mat matpow(Mat A, LL b, int n) {
    Mat res(n, vector<LL>(n,0));
    for (int i=0;i<n;i++) res[i][i]=1;  // 单位矩阵（相当于数字1）
    while (b) {

        A=mul(A,A,n); b>>=1;

    return res;
}

十三、高精度

1. 高精度加法

vector<int> add(vector<int>&A, vector<int>&B) {
    vector<int> C; int t=0;
    for (int i=0;i<(int)A.size()||(int)i<B.size()||t;i++){
        if (i<(int)A.size()) t+=A[i];
        if (i<(int)B.size()) t+=B[i];
        C.push_back(t%10); t/=10;
    }
    return C;
}
// 思路：模拟竖式加法，低位对齐逐位相加，t同时充当当前位之和与进位。
// t%10是当前位结果，t/10是进位。循环条件||t保证最高进位被处理。
// A/B：数字低位在前（个位在[0]）；t：当前位累加值兼进位

2. 高精度减法

bool cmp_big(vector<int>&A, vector<int>&B) {
    if (A.size()!=B.size()) return A.size()>B.size();
    for (int i=A.size()-1;i>=0;i--)
        if (A[i]!=B[i]) return A[i]>B[i];
    return true;
}
vector<int> sub(vector<int>&A, vector<int>&B) {
    vector<int> C; int t=0;
    for (int i=0;i<(int)A.size();i++) {
        t=A[i]-t-(i<(int)B.size()?B[i]:0);
        C.push_back((t+10)%10);
        t=t<0?1:0;
    }
    while (C.size()>1 && C.back()==0) C.pop_back();
    return C;
}
// 思路：模拟竖式减法，t表示借位（0或1）。当前位结果t可能为负，
// (t+10)%10取正确数字，t<0则向高位借1。最后去掉前导零。
// 要求A>=B（用cmp_big判断），若A<B则交换后结果取负。

3. 高精度乘低精度

vector<int> mul(vector<int>&A, int b) {
    vector<int> C; int t=0;
    for (int i=0;i<(int)A.size()||t;i++) {
        if (i<(int)A.size()) t+=A[i]*b;
        C.push_back(t%10); t/=10;
    }
    while (C.size()>1 && C.back()==0) C.pop_back();
    return C;
}
// 思路：高精度数每一位乘以b，累加进位t。与加法类似，t/10是进位。
// b为0时结果应为0，while去掉前导零保证正确。
// A：高精度数（低位在前）；b：普通整数乘数

4. 高精度除低精度

vector<int> div(vector<int>&A, int b, int &r) {
    vector<int> C; r=0;
    for (int i=A.size()-1;i>=0;i--) {
        r=r*10+A[i];
        C.push_back(r/b); r%=b;
    }
    reverse(C.begin(),C.end());
    while (C.size()>1 && C.back()==0) C.pop_back();
    return C;
}
// 思路：模拟竖式除法，从高位到低位处理（A存储低位在前，故从末尾开始）。
// r是当前余数，r*10+A[i]是带入下一位后的被除数，除以b得商和新余数。
// 商存入C后reverse（因为从高位算起），去掉前导零。
// A：高精度数（低位在前）；b：除数；r：余数（输出参数）
