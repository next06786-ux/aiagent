```python
class SkillMatcher:
    """技能匹配分析器"""
    
    def calculate_match_score(
        self,
        user_skills: Dict[str, float],  # {"Python": 7.5, "项目管理": 6.0}
        job_requirements: Dict[str, float]  # {"需求分析": 8.0, "项目管理": 7.0}
    ) -> Dict[str, Any]:
        """
        计算技能匹配度
        
        返回：
        - 总体匹配度
        - 技能缺口
        - 可迁移技能
        - 学习建议
        """
        # 1. 直接匹配的技能
        direct_match = {}
        for skill, required_level in job_requirements.items():
            if skill in user_skills:
                gap = required_level - user_skills[skill]
                direct_match[skill] = {
                    "current_level": user_skills[skill],
                    "required_level": required_level,
                    "gap": gap,
                    "status": "满足" if gap <= 0 else "需提升"
                }
        
        # 2. 可迁移技能（从知识图谱查询相关技能）
        transferable = self._find_transferable_skills(
            user_skills, job_requirements
        )
        
        # 3. 计算总体匹配度
        total_match_score = self._calculate_weighted_match(
            direct_match, transferable, job_requirements
        )
        
        # 4. 生成学习路径
        learning_path = self._generate_learning_path(
            direct_match, job_requirements
        )
        
        return {
            "overall_match": total_match_score,  # 0.72
```

```python
class CareerPathFinder:
    """职业路径搜索器"""
    
    def find_optimal_paths(
        self,
        current_role: str,
        target_role: str,
        max_steps: int = 3
    ) -> List[Dict]:
        """
        找到所有可行的职业路径
        
        基于知识图谱中的真实转换数据
        """
        paths = []
        
        # 从知识图谱查询所有转换关系
        transitions = self._query_career_transitions()
        
        # 动态规划搜索
        for path in self._dp_search(current_role, target_role, transitions, max_steps):
            # 为每条路径计算真实数据
            path_analysis = {
                "path": path,  # ["后端工程师", "技术产品经理", "产品经理"]
                "total_duration_months": self._calculate_duration(path),
                "success_rate": self._calculate_success_rate(path),
                "cost_analysis": self._calculate_costs(path),
                "real_cases": self._find_similar_cases(path)
            }
            paths.append(path_analysis)
        
        return sorted(paths, key=lambda x: x['success_rate'], reverse=True)
    
    def _calculate_duration(self, path: List[str]) -> Dict:
        """基于真实案例计算时间"""
        durations = []
        for i in range(len(path) - 1):
            # 从知识图谱查询真实转换时间
            cases = self._query_transition_cases(path[i], path[i+1])
            avg_duration = sum(c['duration_months'] for c in cases) / len(cases)
            durations.append({
                "from": path[i],
                "to": path[i+1],
                "avg_months": avg_duration,
                "range": [min(c['duration_months'] for c in cases),
                         max(c['duration_months'] for c in cases)],
                "sample_size": len(cases)
            })
        
        return {
            "total_months": sum(d['avg_months'] for d in durations),
            "breakdown": durations
        }
```

```python
class SalaryROICalculator:
    """薪资投资回报率计算器"""
    
    def calculate_roi(
        self,
        current_salary: float,
        target_salary: float,
        learning_cost_hours: int,
        transition_months: int,
        time_horizon_years: int = 5
    ) -> Dict:
        """
        计算职业转换的投资回报率
        
        基于真实薪资数据和时间成本
        """
        # 1. 学习成本
        hourly_cost = current_salary / (40 * 4.33 * 12)  # 时薪
        learning_cost = learning_cost_hours * hourly_cost
        
        # 2. 机会成本（转换期间的收入损失）
        opportunity_cost = current_salary * (transition_months / 12) * 0.3  # 假设30%收入损失
        
        # 3. 总成本
        total_cost = learning_cost + opportunity_cost
        
        # 4. 未来收益（考虑时间折现）
        future_benefits = []
        discount_rate = 0.05  # 5%折现率
        
        for year in range(1, time_horizon_years + 1):
            # 从真实薪资曲线获取预期薪资
            expected_salary = self._get_salary_at_year(target_salary, year)
            salary_gain = expected_salary - self._get_salary_at_year(current_salary, year)
            
            # 时间折现
            discounted_gain = salary_gain / ((1 + discount_rate) ** year)
            future_benefits.append({
                "year": year,
                "salary_gain": salary_gain,
                "discounted_gain": discounted_gain
            })
        
        total_benefit = sum(b['discounted_gain'] for b in future_benefits)
        net_benefit = total_benefit - total_cost
        roi = (net_benefit / total_cost) * 100 if total_cost > 0 else 0
        
        # 5. 回本时间
        cumulative = 0
        payback_years = time_horizon_years
        for b in future_benefits:
            cumulative += b['discounted_gain']
            if cumulative >= total_cost:
                payback_years = b['year']
                break
        
        return {
            "total_cost": total_cost,
            "learning_cost": learning_cost,
            "opportunity_cost": opportunity_cost,
            "total_benefit": total_benefit,
            "net_benefit": net_benefit,
            "roi_percentage": roi,
            "payback_years": payback_years,
            "yearly_breakdown": future_benefits,
            "conclusion": self._generate_roi_conclusion(roi, payback_years)
        }
```

#### 算法4：风险评估（蒙特卡洛模拟）

```python
class RiskAssessor:
    """风险评估器"""
    
    def assess_transition_risk(
        self,
        user_profile: Dict,
        target_role: str,
        num_simulations: int = 10000
    ) -> Dict:
        """
        蒙特卡洛模拟评估转换风险
        
        基于真实数据的概率分布
        """
        success_count = 0
        time_outcomes = []
        salary_outcomes = []
        
        for _ in range(num_simulations):
            # 1. 模拟成功概率（基于真实案例）
            base_success_rate = self._get_base_success_rate(target_role)
            
            # 2. 个人因素调整
            skill_match = user_profile['skill_match_score']
            experience_match = user_profile['experience_match']
            
            adjusted_success_rate = base_success_rate * (
                0.5 + skill_match * 0.3 + experience_match * 0.2
            )
            
            # 3. 随机模拟
            if random.random() < adjusted_success_rate:
                success_count += 1
                
                # 模拟时间（基于真实分布）
                time_mean, time_std = self._get_time_distribution(target_role)
                time_outcome = max(3, random.gauss(time_mean, time_std))
                time_outcomes.append(time_outcome)
                
                # 模拟薪资（基于真实分布）
                salary_mean, salary_std = self._get_salary_distribution(target_role)
                salary_outcome = max(0, random.gauss(salary_mean, salary_std))
                salary_outcomes.append(salary_outcome)
        
        # 统计结果
        return {
            "success_rate": success_count / num_simulations,
            "time_estimate": {
                "mean_months": np.mean(time_outcomes) if time_outcomes else 0,
                "p10": np.percentile(time_outcomes, 10) if time_outcomes else 0,
                "p50": np.percentile(time_outcomes, 50) if time_outcomes else 0,
                "p90": np.percentile(time_outcomes, 90) if time_outcomes else 0
            },
            "salary_estimate": {
                "mean": np.mean(salary_outcomes) if salary_outcomes else 0,
                "p10": np.percentile(salary_outcomes, 10) if salary_outcomes else 0,
                "p50": np.percentile(salary_outcomes, 50) if salary_outcomes else 0,
                "p90": np.percentile(salary_outcomes, 90) if salary_outcomes else 0
            },
            "risk_factors": self._identify_risk_factors(user_profile, target_role),
            "confidence_level": "高" if success_count > 7000 else "中" if success_count > 4000 else "低"
```

```python
class SimilarCaseRetriever:
    """相似案例检索器"""
    
    def __init__(self):
        self.embedding_model = None  # 使用预训练模型，无需训练
        self.case_database = []  # 真实案例库
    
    def find_similar_cases(
        self,
        user_profile: Dict,
        target_role: str,
        top_k: int = 5
    ) -> List[Dict]:
        """
        找到最相似的真实案例
        
        不需要训练模型，使用：
        1. 预训练的文本嵌入模型（如 sentence-transformers）
        2. 向量相似度搜索
        """
        # 1. 将用户画像转换为向量
        user_vector = self._profile_to_vector(user_profile)
        
        # 2. 从案例库中检索相似案例
        similar_cases = []
        for case in self.case_database:
            if case['to_role'] == target_role:
                case_vector = self._profile_to_vector(case['user_profile'])
                similarity = cosine_similarity(user_vector, case_vector)
                
                if similarity > 0.7:  # 相似度阈值
                    similar_cases.append({
                        "case_id": case['case_id'],
                        "similarity": similarity,
                        "from_role": case['from_role'],
                        "to_role": case['to_role'],
                        "success": case['success'],
                        "duration_months": case['duration_months'],
                        "key_actions": case['key_actions'],
                        "challenges": case['challenges'],
                        "lessons_learned": case['lessons_learned'],
                        "why_similar": self._explain_similarity(user_profile, case)
                    })
        
        # 3. 按相似度排序
        similar_cases.sort(key=lambda x: x['similarity'], reverse=True)
        
        return similar_cases[:top_k]
    
    def _explain_similarity(self, user_profile, case):
        """解释为什么相似"""
        reasons = []
        
        if user_profile['years_experience'] - case['user_profile']['years_experience'] < 2:
            reasons.append(f"工作年限相近（都是{user_profile['years_experience']}年左右）")
        
        common_skills = set(user_profile['skills']) & set(case['user_profile']['skills'])
        if len(common_skills) >= 3:
            reasons.append(f"技能背景相似（共同技能：{', '.join(list(common_skills)[:3])}）")
        
        if user_profile['education'] == case['user_profile']['education']:
            reasons.append(f"学历相同（{user_profile['education']}）")
        
        return reasons
```

```python
        risk_factors = []
        
        # 1. 分析失败案例的共同特征
        failed_cases = [c for c in historical_cases if not c['success']]
        success_cases = [c for c in historical_cases if c['success']]
        
        # 2. 技能缺口风险
        skill_gap_risk = self._analyze_skill_gap_risk(
            user_profile, failed_cases, success_cases
        )
        if skill_gap_risk['risk_level'] > 0.5:
            risk_factors.append(skill_gap_risk)
        
        # 3. 年龄/经验风险
        age_risk = self._analyze_age_risk(
            user_profile, failed_cases, success_cases
        )
        if age_risk['risk_level'] > 0.5:
            risk_factors.append(age_risk)
        
        # 4. 财务压力风险
        financial_risk = self._analyze_financial_risk(
            user_profile, target_role
        )
        if financial_risk['risk_level'] > 0.5:
            risk_factors.append(financial_risk)
        
        # 5. 市场时机风险
        market_risk = self._analyze_market_risk(target_role)
        if market_risk['risk_level'] > 0.5:
            risk_factors.append(market_risk)
        
        return sorted(risk_factors, key=lambda x: x['risk_level'], reverse=True)
    
    def _analyze_skill_gap_risk(self, user_profile, failed_cases, success_cases):
        """分析技能缺口风险"""
        # 统计失败案例的平均技能匹配度
        failed_avg_match = np.mean([c['skill_match'] for c in failed_cases])
        success_avg_match = np.mean([c['skill_match'] for c in success_cases])
        
        user_match = user_profile['skill_match_score']
        
        # 如果用户的匹配度接近失败案例的平均值
        if abs(user_match - failed_avg_match) < abs(user_match - success_avg_match):
            risk_level = 0.7
            evidence = f"你的技能匹配度（{user_match:.1%}）接近失败案例的平均水平（{failed_avg_match:.1%}）"
        else:
            risk_level = 0.3
            evidence = f"你的技能匹配度（{user_match:.1%}）接近成功案例的平均水平（{success_avg_match:.1%}）"
        
        return {
            "factor": "技能缺口",
            "risk_level": risk_level,
            "evidence": evidence,
            "mitigation": [
                "重点学习需求分析和原型设计",
                "参与真实项目积累经验",
                "找导师指导"
            ]
        }
```

```python
class TimeEstimator:
    """时间估算器"""
    
    def estimate_transition_time(
        self,
        user_profile: Dict,
        target_role: str,
        historical_cases: List[Dict]
    ) -> Dict:
        """
        估算转换所需时间
        
        使用简单线性回归，数据少也能用
        """
        if len(historical_cases) < 10:
            # 数据太少，使用统计方法
            return self._statistical_estimate(historical_cases)
        
        # 1. 提取特征
        X = []
        y = []
        for case in historical_cases:
            features = [
                case['skill_match'],
                case['years_experience'],
                case['education_level'],
                case['has_mentor'],
                case['full_time_learning']
            ]
            X.append(features)
            y.append(case['duration_months'])
        
        # 2. 简单线性回归
        from sklearn.linear_model import LinearRegression
        model = LinearRegression()
        model.fit(X, y)
        
        # 3. 预测用户的时间
        user_features = [
            user_profile['skill_match_score'],
            user_profile['years_experience'],
            user_profile['education_level'],
            1 if user_profile.get('has_mentor') else 0,
            1 if user_profile.get('full_time_learning') else 0
        ]
        
        predicted_months = model.predict([user_features])[0]
        
        # 4. 计算置信区间
        residuals = [abs(y[i] - model.predict([X[i]])[0]) for i in range(len(y))]
        std_error = np.std(residuals)
        
        return {
            "estimated_months": round(predicted_months, 1),
            "confidence_interval": {
                "lower": round(predicted_months - 1.96 * std_error, 1),
                "upper": round(predicted_months + 1.96 * std_error, 1)
            },
            "factors_impact": {
                "skill_match": f"{'加速' if user_profile['skill_match_score'] > 0.7 else '减速'} {abs(model.coef_[0] * 0.1):.1f}个月",
                "experience": f"{'加速' if user_profile['years_experience'] > 3 else '减速'} {abs(model.coef_[1] * 0.5):.1f}个月",
                "mentor": f"有导师可加速 {abs(model.coef_[3]):.1f}个月" if user_profile.get('has_mentor') else "建议找导师"
            },
            "based_on": f"{len(historical_cases)}个真实案例",
            "model_accuracy": f"预测误差 ±{std_error:.1f}个月"
        }
```

```python
class ScenarioValidator:
    """情境验证器"""
    
    def validate_and_annotate(
        self,
        scenario_map: List[Dict],
        algorithm_results: Dict,
        similar_cases: List[Dict]
    ) -> List[Dict]:
        """
        验证情境的合理性，并标注数据来源
        """
        validated_scenarios = []
        
        for scenario in scenario_map:
            # 1. 验证时间线合理性
            time_validation = self._validate_timeline(
                scenario, algorithm_results['path_analysis']
            )
            
            # 2. 验证技能进度
            skill_validation = self._validate_skill_progress(
                scenario, algorithm_results['skill_analysis']
            )
            
            # 3. 验证成本估算
            cost_validation = self._validate_cost(
                scenario, algorithm_results['roi_analysis']
            )
            
            # 4. 标注数据来源
            annotated_scenario = self._annotate_sources(
                scenario, algorithm_results, similar_cases
            )
            
            # 5. 添加验证标记
            annotated_scenario['validation'] = {
                'timeline_valid': time_validation['valid'],
                'skill_progress_valid': skill_validation['valid'],
                'cost_valid': cost_validation['valid'],
                'confidence': self._calculate_confidence([
                    time_validation, skill_validation, cost_validation
                ]),
                'warnings': time_validation.get('warnings', []) + 
                           skill_validation.get('warnings', []) +
                           cost_validation.get('warnings', [])
            }
            
            validated_scenarios.append(annotated_scenario)
        
        return validated_scenarios
    
    def _validate_timeline(self, scenario, path_analysis):
        """验证时间线是否符合算法预测"""
        month = scenario['month']
        phase = scenario['phase']
        
        expected_duration = path_analysis[0]['total_duration_months']['total_months']
        
        warnings = []
        
        # 检查阶段划分是否合理
        if month <= 3 and phase != "技能学习期":
            warnings.append(f"第{month}月应该在技能学习期，但标记为{phase}")
        
        if month > expected_duration:
            warnings.append(f"超出预期时长（{expected_duration}个月）")
        
        return {
            'valid': len(warnings) == 0,
            'warnings': warnings
        }
    
    def _annotate_sources(self, scenario, algorithm_results, similar_cases):
        """标注每个情境的数据来源"""
        
        # 自动识别数据来源
        sources = []
        evidence = []
        
        # 检查是否引用了技能分析
        for skill_gap in algorithm_results['skill_analysis']['skill_gaps']:
            if skill_gap['skill'] in str(scenario):
                sources.append('技能匹配分析')
                evidence.append(f"技能缺口：{skill_gap['skill']}需提升{skill_gap['gap']:.1f}分")
        
        # 检查是否引用了案例
        for case in similar_cases:
            if case['case_id'] in str(scenario.get('real_case_reference', '')):
                sources.append(f"真实案例#{case['case_id']}")
                evidence.append(f"案例相似度{case['similarity']:.1%}")
        
        # 检查是否引用了风险评估
        for risk in algorithm_results['risk_analysis']['risk_factors']:
            if risk['factor'] in str(scenario):
                sources.append('风险评估')
                evidence.append(f"风险因素：{risk['factor']}")
        
        scenario['data_provenance'] = {
            'sources': list(set(sources)),
            'evidence': evidence,
            'confidence': len(sources) / 3,  # 至少3个来源才高置信度
            'traceable': len(sources) > 0
        }
        
        return scenario
```

```
- 证据：技能缺口需求分析2.5分，案例#001在第3个月也遇到此问题
```

```python
from dataclasses import dataclass

from backend.knowledge_graph.neo4j_client import Neo4jClient


@dataclass
class DataDrivenScenario:
    """数据驱动的情境节点"""
    month: int
    phase: str
    scenario: str
    potential_challenges: List[str]
    potential_opportunities: List[str]
    attention_areas: Dict[str, str]
    depends_on: List[str]
    data_sources: List[str]

    # 数据溯源

    # 验证信息
    validation: Dict[str, Any]


class DataDrivenDecisionEngine:
    """数据驱动的决策引擎"""
    
    def __init__(self):
        # 算法层
        self.career_algo = CareerDecisionAlgorithm()
        self.kg_integration = None  # 延迟初始化
        
        # LLM层
        self.lora_analyzer = LoRADecisionAnalyzer()
        
        # 数据层
        self.neo4j_client = Neo4jClient()
        
    async def generate_data_driven_scenario_map(
        self,
        user_id: str,
        question: str,
        option: Dict[str, str],
        collected_info: Dict[str, Any]
    ) -> Dict[str, Any]:
        """
        生成数据驱动的情境地图
        
        完整流程：
        1. 从知识图谱提取用户数据
        2. 运行算法分析
        3. 检索相似案例
        4. 生成增强提示词
        5. LLM生成情境
        6. 验证和标注
        """
        
        print(f"\n{'='*60}")
        print(f"开始生成数据驱动的决策推演")
        print(f"问题：{question}")
        print(f"选项：{option['title']}")
        print(f"{'='*60}\n")
        
        # ========== 第一步：提取用户数据 ==========
        print("📊 第一步：从知识图谱提取用户数据...")
        
        self.kg_integration = KnowledgeGraphCareerIntegration(user_id)
        personal_capital = self.kg_integration.extract_personal_capital_from_kg()
        career_graph = self.kg_integration.build_career_graph_from_real_data()
        
        print(f"  ✓ 提取到 {len(personal_capital.skills)} 个技能")
        print(f"  ✓ 工作经验：{personal_capital.years_experience} 年")
        print(f"  ✓ 人脉规模：{personal_capital.network_size} 人")
        
        # ========== 第二步：运行算法分析 ==========

        skill_analysis = await self._analyze_skill_match(
            user_id, option['title'], personal_capital
        )
        print(f"  ✓ 技能匹配度：{skill_analysis['overall_match']:.1%}")
        
        # 2.2 职业路径分析
        path_analysis = self._analyze_career_path(
            collected_info.get('current_role', '后端工程师'),
            option['title'],
            career_graph
        )
        print(f"  ✓ 找到 {len(path_analysis)} 条可行路径")
        
        # 2.3 ROI分析
        roi_analysis = self._analyze_roi(
            collected_info, option, skill_analysis, path_analysis
        )
        print(f"  ✓ ROI：{roi_analysis['roi_percentage']:.1f}%")
        
        # 2.4 风险评估
        risk_analysis = self._assess_risk(
            personal_capital, option, skill_analysis
        )

    async def _analyze_skill_match(
        self,
        user_id: str,
        target_role: str,
        personal_capital
    ) -> Dict:
        """技能匹配分析"""
        
        # 从知识图谱获取目标岗位要求
        job_requirements = self.neo4j_client.query(f"""
            MATCH (j:Job {{title: '{target_role}'}})-[r:REQUIRES_SKILL]->(s:Skill)
            RETURN s.name as skill, r.min_level as required_level, r.importance as weight
        """)
        
        # 用户当前技能
        user_skills = {
            skill.name: skill.level 
            for skill in personal_capital.skills
        }
        
        # 计算匹配度
        direct_match = {}
        skill_gaps = []
        
        for req in job_requirements:
            skill_name = req['skill']
            required_level = req['required_level']
            weight = req['weight']
            
            if skill_name in user_skills:
                current_level = user_skills[skill_name]
                gap = required_level - current_level
                
                direct_match[skill_name] = {
                    'current_level': current_level,
                    'required_level': required_level,
                    'gap': gap,
                    'weight': weight,
                    'status': '满足' if gap <= 0 else '需提升'
                }
                
                if gap > 0:
                    skill_gaps.append({
                        'skill': skill_name,
                        'gap': gap,
                        'priority': '高' if weight > 0.7 else '中' if weight > 0.4 else '低',
                        'learning_time_hours': int(gap * 40),  # 每0.1分需要4小时
                        'resources': self._get_learning_resources(skill_name)
                    })
            else:
                # 完全缺失的技能
                skill_gaps.append({
                    'skill': skill_name,
                    'gap': required_level,
                    'priority': '高' if weight > 0.7 else '中',
                    'learning_time_hours': int(required_level * 40),
                    'resources': self._get_learning_resources(skill_name)
                })
        
        # 计算总体匹配度
        if job_requirements:
            matched_weight = sum(
                req['weight'] for req in job_requirements
                if req['skill'] in user_skills and 
                user_skills[req['skill']] >= req['required_level']
            )
            total_weight = sum(req['weight'] for req in job_requirements)
            overall_match = matched_weight / total_weight if total_weight > 0 else 0
        else:
            overall_match = 0.5
        
        # 估算学习时间
        total_learning_hours = sum(gap['learning_time_hours'] for gap in skill_gaps)
        estimated_time_months = total_learning_hours / 160  # 每月160小时
        
        return {
            'overall_match': overall_match,
            'direct_match': direct_match,
            'skill_gaps': sorted(skill_gaps, key=lambda x: x['learning_time_hours'], reverse=True),
            'estimated_time_months': estimated_time_months,
            'total_learning_hours': total_learning_hours,
            'data_source': 'knowledge_graph + algorithm'
        }

        )

        for path, value in [paths]:  # 只取最优路径
            # 从知识图谱查询转换数据
            transitions = []
            for i in range(len(path) - 1):
                from_role = path[i]
                to_role = path[i + 1]
                
                # 查询真实转换案例
                cases = self.neo4j_client.query(f"""
                    MATCH (c:CaseStudy)
                    WHERE c.from_role = '{from_role}' AND c.to_role = '{to_role}'
                    RETURN c.duration_months as duration, c.success as success
                """)
                
                if cases:
                    avg_duration = sum(c['duration'] for c in cases) / len(cases)
                    success_rate = sum(1 for c in cases if c['success']) / len(cases)
                else:
                    avg_duration = 12
                    success_rate = 0.5
                
                transitions.append({
                    'from': from_role,
                    'to': to_role,
                    'avg_duration_months': avg_duration,
                    'success_rate': success_rate,
                    'sample_size': len(cases)
                })
            
            enriched_paths.append({
                'path': path,
                'value': value,
                'transitions': transitions,
                'total_duration_months': {
                    'total_months': sum(t['avg_duration_months'] for t in transitions),
                    'breakdown': transitions
                },
                'success_rate': min(t['success_rate'] for t in transitions) if transitions else 0.5,
                'real_cases': {
                    'sample_size': sum(t['sample_size'] for t in transitions)
                }
            })
        
        return enriched_paths
```
